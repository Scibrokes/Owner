## 下载OpenVPN代理服务器
wget https://git.io/vpn -O openvpn-install.sh

## 授权执行权限
sudo chmod +x openvpn-install.sh

## 安装OpenVPN代理服务器
sudo bash openvpn-install.sh

## 停止运行OpenVPN代理服务器
sudo systemctl stop openvpn-server@server.service
